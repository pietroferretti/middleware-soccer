103 int readInterruptionEvent(FILE **file, struct interruption_event *new, picoseconds start) {
104 
105  // temporary variables to parse the file
106  picoseconds minutes;
107  double seconds;
108 
109  // read one line from the referee events file
110  int read = fscanf(*file, "%*31c:%lu:%lf;%*s\n", &minutes, &seconds);
111 
112  if (!read) {
113  // return -1 to signal that the file has ended
114  return -1;
115  } else {
116  // we manage to read one event (the start of an interruption)
117  new->start = start + (picoseconds) (seconds * SECTOPIC) + (minutes * 60) * SECTOPIC;
118  DBG(("\nSTART INTERR %lu,%f", minutes, seconds));
119 
120  // get the corresponding end of the interruption
121  fscanf(*file, "%*29c:%lu:%lf;%*s\n", &minutes, &seconds);
122  new->end = start + (picoseconds) (seconds * SECTOPIC) + (minutes * 60) * SECTOPIC;
123  DBG(("\nEND INTERR %lu,%f", minutes, seconds));
124  return 0;
125  }
